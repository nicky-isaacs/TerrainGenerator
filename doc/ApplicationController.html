def cache_file(key, data, extension=nil)
  begin FileUtils.mkdir_p(Rails.root, 'tmp', 'obj_caching'); rescue; end
  path = File.join(Rails.root, 'tmp', 'obj_caching', key)
  path += extension if extension

  file = File.new path, "w+"
  file << data
  file.close
  path
end
